#git操作流程
git init #初始化git  
git add . #将本地代码提交到暂存区  
git commit -m "first commit" #提交到本地仓库  
git remote add origin https://github.com/ideshun/site-navigation.git #添加远程仓库  
git push -u origin master #将本地代码提交到远程仓库的master分支  
git status #查看暂存区  
git rm -f 文件名  #删除暂存区文件
